java在arcgis地图上实现打点以及弹框

使用arcgis api for javascript 4.16 实现在地图上打点和显示点是弹框，以下为效果图。

源代码：

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>在地图上打点并显示弹框</title>
    <link rel="stylesheet" href="https://js.arcgis.com/4.16/esri/css/main.css">
    <script src="https://cdn.bootcss.com/jquery/3.2.1/jquery.min.js"></script>
    <script src="https://js.arcgis.com/4.16/init.js"></script>
    <style>
        #viewDiv{
            width: 100%;
            height: 100%;
        }
    </style>
</head>
<body style="overflow: hidden">
<div id="viewDiv" style="background-color:lightskyblue" class="col-sm-6"></div>
</body>
</html>
<script>
    require([
        "esri/Map",
        "esri/views/MapView",
        "esri/views/SceneView",
        "esri/layers/FeatureLayer",
    ],function (Map,MapView,SceneView,FeatureLayer) {
        var featureLayer;
        //构建2D底图
        var mainMap = new Map({
            basemap:"satellite",
        });
        //创建MapView来存放各个图层
        var mainView = new MapView({
            map:mainMap,
            container:"viewDiv",
            center:[108.94878,34.22259],
            zoom:13,
        });
        //将JSON数据封装成点
        function CreatePoint(pointMes) {
            //点对应的JSON数据就放到attributes属性中
            var feature = {
                geometry: {
                    type: "point",
                    longitude: pointMes.lng,
                    latitude: pointMes.lat,
                },
                attributes: {
                    "location": pointMes.location,
                    "area": pointMes.area,
                    "num": pointMes.num,
                    "level":pointMes.level,
                    "district": pointMes.district,
                    "ObjectID":pointMes.id,
                }
            };
            return feature
        }
        //弹框
        var popupTemplate = {
            title: "<p><b>{location}位置</b> 详情</p>",
            content: [
                {
                    type: "fields",
                    fieldInfos: [
                        {
                            fieldName: "location", //attributes key值
                            label: "位置", //popupTemplate中显示的key名
                            visible: true //该属性字段是否在popupTemplate可见
                        },
                        {
                            fieldName: "district",
                            label: "所属区",
                            visible: true
                        },
                        {
                            fieldName: "level",
                            label: "等级",
                            visible: true
                        }
                    ]
                },
            ]
        };
        function createFillSymbol(value, color,url,label) {
            return {
                value: value,
                symbol: {
                    type: "picture-marker",
                    url: url,
                    width: "40px",
                    height: "40px"
                },
                label: label,
                font: {
                    size: "15px",
                    family: "Noto Sans",
                    style: "italic",
                    weight: "normal"
                }
            };
        }
        var renderDam = {
            type: "unique-value",
            field: "level",
            uniqueValueInfos: [
                createFillSymbol("level1", "#d81e06","images/BluePin1LargeB.png","一级"),
                createFillSymbol("level2", "#ffa200","images/GreenPin1LargeB.png","二级"),
                createFillSymbol("level3", "#f5f21d","images/RedPin1LargeB.png","三级）"),
            ]
        };
        //这里发送请求获得JSON数据，遍历生成point点，并添加到图层中进行显示
        function Query() {
            var features = [];
            //元素不重复的数组
            var damSeries = new Set();
            $.ajax({
                type: "GET",
                url: '/getPoints',
                async:false,
                success: function (response) {
                    console.log("输出点"+response);
                    for (let pointMes of response) {
                        features.push(CreatePoint(pointMes));  //在地图上打点
                        damSeries.add(pointMes.location)  //
                    }
                    //创建图层来存放Graphics,处理为chart显示需要的数据
                    featureLayer = new FeatureLayer({
                        fields: [
                            {
                                name: "location",
                                alias: "location",
                                type: "string"
                            }, {
                                name: "area",
                                alias: "area",
                                type: "integer"
                            }, {
                                name: "num",
                                alias: "num",
                                type: "integer"
                            }, {
                                name: "level",
                                alias: "level",
                                type: "string"
                            }, {
                                name: "district",
                                alias: "district",
                                type: "string"
                            }
                        ],
                        objectIdField: "ObjectID",
                        geometryType: "point",
                        source: features,
                        popupTemplate: popupTemplate,
                        outFields: ["*"],
                        renderer: renderDam,                //图例
                        labelingInfo:[
                            {
                                labelExpressionInfo:{
                                    value:"{district}"
                                },
                                symbol:{
                                    type:"text",
                                    color: "white",
                                    haloSize: 1,
                                    //haloColor: "blue",
                                    font: {  // autocast as new Font()
                                        family: "Ubuntu Mono",
                                        size: 14, 
                                    }
                                }
                            }
                        ]
                    });
                }
            });}
        Query();
        //直接将打点的图层放在地图上
        mainMap.layers.add(featureLayer)
    })
</script>
