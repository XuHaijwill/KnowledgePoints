arcgis api（三）arcgis api for js 4.x 加载高德地图、谷歌地图、天地图

  arcgis api for 4.x支持2D和3D地图的展示，3.x的版本我们通过自定义Layer实现了高德地图、谷歌地图和天地图的加载。本文介绍了，在线地图服务、高德地图、谷歌地图、天地图在4.x版本中的加载方法。

1、在线地图服务

  1. import React, { Component } from "react";
  2. import esriLoader from "esri-loader";
  3.  
  4. //在线切片底图服务
  5. export default class GisServerMap extends Component {
  6.   constructor(props) {
  7.     super(props);
  8.     this.tileMapUrl =
  9.       "http://map.geoq.cn/ArcGIS/rest/services/ChinaOnlineStreetPurplishBlue/MapServer";
  10.   }
  11.   componentDidMount() {
  12.     this.initMap();
  13.   }
  14.   initMap() {
  15.     const mapURL = {
  16.       url: "https://js.arcgis.com/4.7/"
  17.     };
  18.     esriLoader
  19.       .loadModules(
  20.         [
  21.           "esri/Map",
  22.           "esri/Basemap",
  23.           "esri/layers/TileLayer",
  24.           "esri/views/MapView",
  25.           "esri/views/SceneView",
  26.           "dojo/domReady!"
  27.         ],
  28.         mapURL
  29.       )
  30.       .then(([Map, Basemap, TileLayer, MapView, SceneView]) => {
  31.         let layer = new TileLayer({
  32.           url: this.tileMapUrl
  33.         });
  34.         //实例化底图对象
  35.         let baseMap = new Basemap({
  36.           baseLayers: [layer],
  37.           title: "Custom Basemap",
  38.           id: "myBasemap"
  39.         });
  40.         //实例化地图对象
  41.         let map = new Map({
  42.           basemap: baseMap
  43.         });
  44.         //实例化view
  45.         //2D
  46.         let view2D = new MapView({
  47.           center: [120.2, 32.1],
  48.           map: map,
  49.           container: "mapDiv2D",
  50.           zoom: 5
  51.         });
  52.         //3D
  53.         let view3D = new SceneView({
  54.           center: [120.2, 32.1],
  55.           map: map,
  56.           container: "mapDiv3D",
  57.           zoom: 1
  58.         });
  59.       });
  60.   }
  61.   render() {
  62.     let style = {
  63.       width: "50%",
  64.       height: "100%",
  65.       float: 'left'
  66.     };
  67.     return (
  68.       <div id="GisServerMap" className="mapContent">
  69.         <div id="mapDiv2D" style={style} />
  70.         <div id="mapDiv3D" style={style} />
  71.         <div style={{clear: 'both'}} />
  72.       </div>
  73.     );
  74.   }
  75. }

  31.         mapURL
  32.       )
  33.       .then(
  34.         ([
  35.           Map,
  36.           esriConfig,
  37.           esriRequest,
  38.           Color,
  39.           SceneView,
  40.           LayerList,
  41.           BaseTileLayer,
  42.           MapView
  43.         ]) => {
  44.           // *******************************************************
  45.           // Custom tile layer class code
  46.           // Create a subclass of BaseTileLayer
  47.           // *******************************************************
  48.  
  49.           let TintLayer = BaseTileLayer.createSubclass({
  50.             properties: {
  51.               urlTemplate: null,
  52.               tint: {
  53.                 value: null,
  54.                 type: Color
  55.               }
  56.             },
  57.  
  58.             // generate the tile url for a given level, row and column
  59.             getTileUrl: function(level, row, col) {
  60.               return this.urlTemplate
  61.                 .replace("{z}", level)
  62.                 .replace("{x}", col)
  63.                 .replace("{y}", row);
  64.             },
  65.  
  66.             // This method fetches tiles for the specified level and size.
  67.             // Override this method to process the data returned from the server.
  68.             fetchTile: function(level, row, col) {
  69.               // call getTileUrl() method to construct the URL to tiles
  70.               // for a given level, row and col provided by the LayerView
  71.               var url = this.getTileUrl(level, row, col);
  72.  
  73.               // request for tiles based on the generated url
  74.               // set allowImageDataAccess to true to allow
  75.               // cross-domain access to create WebGL textures for 3D.
  76.               return esriRequest(url, {
  77.                 responseType: "image",
  78.                 allowImageDataAccess: true
  79.               }).then(
  80.                 function(response) {
  81.                   // when esri request resolves successfully
  82.                   // get the image from the response
  83.                   var image = response.data;
  84.                   var width = this.tileInfo.size[0];
  85.                   var height = this.tileInfo.size[0];
  86.  
  87.                   // create a canvas with 2D rendering context
  88.                   var canvas = document.createElement("canvas");
  89.                   var context = canvas.getContext("2d");
  90.                   canvas.width = width;
  91.                   canvas.height = height;
  92.  
  93.                   // Draw the blended image onto the canvas.
  94.                   context.drawImage(image, 0, 0, width, height);
  95.  
  96.                   return canvas;
  97.                 }.bind(this)
  98.               );
  99.             }
  100.           });
  101.  
  102.           // *******************************************************
  103.           // Start of JavaScript application
  104.           // *******************************************************
  105.  
  106.           // Add stamen url to the list of servers known to support CORS specification.
  107.           esriConfig.request.corsEnabledServers.push("webst01.is.autonavi.com");
  108.  
  109.           // Create a new instance of the TintLayer and set its properties
  110.           let digitallTileLayer = new TintLayer({
  111.             urlTemplate: 'http://webst01.is.autonavi.com/appmaptile?lang=zh_cn&size=1&scale=1&style=8&x={x}&y={y}&z={z}',
  112.             tint: new Color("#004FBB"),
  113.             title: "高德"
  114.           });
  115.           let satelliteTileLayer = new TintLayer({
  116.             urlTemplate: 'http://webst01.is.autonavi.com/appmaptile?style=6&x={x}&y={y}&z={z}',
  117.             tint: new Color("#004FBB"),
  118.             title: "高德"
  119.           });
  120.           let stamenTileLayer = new TintLayer({
  121.             urlTemplate: 'http://webst01.is.autonavi.com/appmaptile?style=8&x={x}&y={y}&z={z}',
  122.             tint: new Color("#004FBB"),
  123.             title: "高德"
  124.           });
  125.  
  126.           // add the new instance of the custom tile layer the map
  127.           let map = new Map({
  128.             layers: [satelliteTileLayer, stamenTileLayer]
  129.           });
  130.  
  131.           // create a new scene view and add the map
  132.           //2d
  133.           let view2D = new MapView({
  134.             container: "mapDiv2D",
  135.             map: map,
  136.             center: [110.1, 23.8],
  137.             zoom: 3
  138.           });
  139.           //3d
  140.           let view3D = new SceneView({
  141.             container: "mapDiv3D",
  142.             map: map,
  143.             center: [110.1, 23.8],
  144.             zoom: 1          
  145.           })
  146.         }
  147.       );
  148.   }
  149.   render() {
  150.     let style = {
  151.       width: "50%",
  152.       height: "100%",
  153.       float: 'left'
  154.     };
  155.     return (
  156.       <div id="GisAMap" className="mapContent">
  157.         <div id="mapDiv2D" style={style} />
  158.         <div id="mapDiv3D" style={style} />
  159.         <div style={{clear: 'both'}} />
  160.       </div>
  161.     );
  162.   }
  163. }
