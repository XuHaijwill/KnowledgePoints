原生支持的数据库：

怎么才能支持国产的一些数据库呢，废话不多说，直接上代码。

  1. 1、流程引擎扩展配置接口
  2. public interface ActivitiEngineConfigurationConfigurer {
  3.     public void configure(SpringProcessEngineConfiguration processEngineConfiguration);
  4. }
  5.  
  6. 扩展实现国产数据库类型可配置
  7. @Component
  8. public class ActivitiEngineConfiguration implements ActivitiEngineConfigurationConfigurer {
  9.  
  10.     //设置数据库类型
  11.     @Value(value = "${jiuqi.nr.activiti.dbtype}")
  12.     private String databaseType;
  13.  
  14.     //设置DbSqlSessionConfigurator
  15.     @Autowired
  16.     private DbSqlSessionConfigurator dbSqlSessionConfigurator;
  17.  
  18.     @Override
  19.     public void configure(SpringProcessEngineConfiguration processEngineConfiguration) {
  20.         //设置数据库类型
  21.         processEngineConfiguration.setDatabaseType(databaseType);
  22.         processEngineConfiguration.setUsingRelationalDatabase(false);
  23.         List<ProcessEngineConfigurator> configurators = new ArrayList<ProcessEngineConfigurator>();
  24.         configurators.add(dbSqlSessionConfigurator);
  25.         processEngineConfiguration.setConfigurators(configurators);
  26.  
  27.  
  28.     }
  29. }
  30.  
  31. 2、编写ProcessEngineConfigurator，配置SqlSessionFactory
  32. /**
  33.  * SqlSessionFactory配置
  34.  */
  35. @Component
  36. public class DbSqlSessionConfigurator implements ProcessEngineConfigurator {
  37.  
  38.     private static final String DEFAULT_MYBATIS_MAPPING_FILE = "config/mapper/mappings.xml";
  39.  
  40.     private static final String DATABASE_GBASE="gbase";
  41.  
  42.     private static final String DATABASE_DM="dameng";
  43.  
  44.     private static final String DATABASE_GAUSSDB100="gaussdb100";
  45.  
  46.     private static final String DATABASE_OSCAR="oscar";
  47.  
  48.     private  static Set<String> dbSets=new HashSet<>();
  49.  
  50.     static {
  51.         dbSets.add(DATABASE_GBASE);
  52.         dbSets.add(DATABASE_DM);
  53.         dbSets.add(DATABASE_GAUSSDB100);
  54.         dbSets.add(DATABASE_OSCAR);
  55.     }
  56.  
  57.     /**此出设置重写的NrDbSqlSessionFactory是为了处理Gbase8s不支持批量插入，其他数据库可不实现该方法*/
  58.     @Override
  59.     public void beforeInit(ProcessEngineConfigurationImpl processEngineConfiguration) {
  60.         //添加重写的SqlSessionFactory
  61.         processEngineConfiguration.setDbSqlSessionFactory(new NrDbSqlSessionFactory());
  62.     }
  63.  
  64.     //ProcessEngineConfigurationImpl中的configuratorsAfterInit会执行到此处
  65.     @Override
  66.     public void configure(ProcessEngineConfigurationImpl processEngineConfiguration) {
  67.         //创建SqlSessionFactory并设置ProcessEngineConfigurationImpl
  68.         processEngineConfiguration.setSqlSessionFactory(createSqlSessionFactory(processEngineConfiguration));
  69.         //重新执行initDbSqlSessionFactory
  70.         processEngineConfiguration.initDbSqlSessionFactory();
  71.  
  72.     }
  73.  
  74.     private SqlSessionFactory createSqlSessionFactory(ProcessEngineConfigurationImpl processEngineConfiguration) {
  75.         InputStream inputStream = null;
  76.         try {
  77.             //读取的mappings.xm为了处理gbase8s的长字符串类型错误，不是gbase可以加载activiti默认的配置
  78.             inputStream = getResourceAsStream(DEFAULT_MYBATIS_MAPPING_FILE);
  79.  
  80.             Environment environment = new Environment("default", processEngineConfiguration.getTransactionFactory(),
  81.                     processEngineConfiguration.getDataSource());
  82.             Reader reader = new InputStreamReader(inputStream);
  83.             Properties properties = new Properties();
  84.             properties.put("prefix", processEngineConfiguration.getDatabaseTablePrefix());
  85.             String wildcardEscapeClause = "";
  86.             if ((processEngineConfiguration.getDatabaseWildcardEscapeCharacter() != null)
  87.                     && (processEngineConfiguration.getDatabaseWildcardEscapeCharacter().length() != 0)) {
  88.                 wildcardEscapeClause = " escape '" + processEngineConfiguration.getDatabaseWildcardEscapeCharacter()
  89.                         + "'";
  90.             }
  91.             properties.put("wildcardEscapeClause", wildcardEscapeClause);
  92.             // set default properties
  93.             properties.put("limitBefore", "");
  94.             properties.put("limitAfter", "");
  95.             properties.put("limitBetween", "");
  96.             properties.put("limitOuterJoinBetween", "");
  97.             properties.put("limitBeforeNativeQuery", "");
  98.             properties.put("orderBy", "order by ${orderByColumns}");
  99.             properties.put("blobType", "BLOB");
  100.             properties.put("boolValue", "TRUE");
  101.  
  102.             if (processEngineConfiguration.getDatabaseType() != null) {
  103.                 //设置读取数据库语法配置文件，文件路径src/main/resource/config/custom/   自定义的数据类型对应语法文件
  104.                 if(dbSets.contains(processEngineConfiguration.getDatabaseType().toLowerCase())) {
  105.                     properties.load(getResourceAsStream("config/custom/"+ processEngineConfiguration.getDatabaseType() + ".properties"));
  106.                 }else {
  107.                     properties.load(getResourceAsStream("org/activiti/db/properties/"
  108.                             + processEngineConfiguration.getDatabaseType() + ".properties"));
  109.                 }
  110.  
  111.             }
  112.  
  113.             Configuration configuration = processEngineConfiguration.initMybatisConfiguration(environment, reader,
  114.                     properties);
  115.             return new DefaultSqlSessionFactory(configuration);
  116.  
  117.         } catch (Exception e) {
  118.             throw new ActivitiException("Error while building ibatis SqlSessionFactory: " + e.getMessage(), e);
  119.         } finally {
  120.             IoUtil.closeSilently(inputStream);
  121.         }
  122.  
  123.     }
  124.  
  125.     private InputStream getResourceAsStream(String resource) {
  126.         return ReflectUtil.getResourceAsStream(resource);
  127.     }
  128.  
  129.     @Override
  130.     public int getPriority() {
  131.         return 0;
  132.     }
  133. }
  134.  
  135. 2、重写DbSqlSession的flushInsertEntities方法(适配gbase才需要此步骤)
  136. /**
  137.  * SqlSession会话工厂
  138.  * 在DbSqlSessionConfigurator（beforeInit方法）中重置会话工厂
  139.  */
  140. public class MyDbSqlSessionFactory extends DbSqlSessionFactory {
  141.  
  142.     @Override
  143.     public Session openSession(CommandContext commandContext) {
  144.         return new MyDbSqlSession(this, commandContext.getEntityCache());
  145.     }
  146.  
  147. }
  148.  
  149.  
  150. public class NrDbSqlSession extends DbSqlSession {
  151.  
  152.     private static final Logger log = LoggerFactory.getLogger(NrDbSqlSession.class);
  153.  
  154.     public NrDbSqlSession(DbSqlSessionFactory dbSqlSessionFactory, EntityCache entityCache) {
  155.         super(dbSqlSessionFactory, entityCache);
  156.     }
  157.  
  158.     /**
  159.      * 重写此方法是为了解决gbase不支持批量插入的问题
  160.      * @param entityClass
  161.      * @param entitiesToInsert
  162.      */
  163.     @Override
  164.     protected void flushInsertEntities(Class<? extends Entity> entityClass, Collection<Entity> entitiesToInsert) {
  165.         if (entitiesToInsert.size() == 1) {
  166.             flushRegularInsert(entitiesToInsert.iterator().next(), entityClass);
  167.         }else {
  168.             for (Entity entity : entitiesToInsert) {
  169.                 flushRegularInsert(entity, entityClass);
  170.             }
  171.         }
  172.     }
  173.  
  174. }
  175.  
  176. 3、装配自定义扩展配置
  177. MyProcessEngineConfiguration cfg = new MyProcessEngineConfiguration();
  178. engineFactory.setProcessEngineConfiguration(cfg);
  179. buildDataSource(cfg);
  180. //判断原生activiti是否支持应用数据库
  181. if(!isSupportDatabase(cfg)){
  182.     //国产数据库扩展适配，构建引擎之前
  183.    activitiEngineConfigurationConfigurer.configure(cfg);
  184. }
  185. return new Activiti6ProcessEngine(engineFactory.getObject()).setActorStrategyProvider(actorStrategyProvider)
  186.       .setUserActionEventHandler(actionEventHandler).setProcessProviders(processProviders)
  187.       .setNrParameterUtils(ParameterUtils).setConditionalExecute(conditionalExecutes);

最后说一句，不按以上方法也可以，可以重写ProcessEngineConfigurationImpl类的相关方法实现。当然，你也可以修改activiti源码，替换jar包，但是那样很不方便也不太合适。
